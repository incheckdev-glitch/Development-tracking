// <- keep in sync with frontend /** Utilities **/ function _success(obj) { return ContentService.createTextOutput( JSON.stringify(Object.assign({ ok: true }, obj)) ).setMimeType(ContentService.MimeType.JSON); } function _error(msg) { return ContentService.createTextOutput( JSON.stringify({ ok: false, error: msg }) ).setMimeType(ContentService.MimeType.JSON); } function _normalizeDate(v) { if (!v) return ""; let s = String(v).trim(); // If already Date type if (Object.prototype.toString.call(v) === "[object Date]" && !isNaN(v)) { return Utilities.formatDate(v, Session.getScriptTimeZone(), "yyyy-MM-dd"); } // ISO (keep date part) if (s.indexOf("T") > -1 && /^\d{4}-\d{2}-\d{2}T/.test(s)) { return s.substring(0, 10); } // yyyy-mm-dd if (/^\d{4}-\d{2}-\d{2}$/.test(s)) return s; // mm/dd/yyyy if (/^\d{1,2}\/\d{1,2}\/\d{4}$/.test(s)) { const [mm, dd, yy] = s.split("/"); return Utilities.formatString("%s-%02d-%02d", yy, Number(mm), Number(dd)); } // dd/mm/yyyy if (/^\d{1,2}\/\d{1,2}\/\d{4}$/.test(s)) { const [dd, mm, yy] = s.split("/"); return Utilities.formatString("%s-%02d-%02d", yy, Number(mm), Number(dd)); } // Fallback: try Date parse const d = new Date(s); if (!isNaN(d)) { return Utilities.formatDate(d, Session.getScriptTimeZone(), "yyyy-MM-dd"); } return ""; } /** doGet: returns normalized objects with consistent keys */ function doGet(e) { try { const secret = e.parameter.secret;
